% #3 : Label

	\begin{figure}[h]
		\centering
		\includegraphics[width=0.9\textwidth]{#1}

		\caption{#2}
		\label{#3}
	\end{figure}
}

% Befehl \code{Quelldatei}{Beschriftung}{label}{Sprache}
\def\code#1#2#3#4{ %
	\begin{center}
	\centering
	\lstset{language=#4,caption={#2},label=#3}
	\singlespacing
	\lstinputlisting[language=#4]{#1}
	\end{center}
}
